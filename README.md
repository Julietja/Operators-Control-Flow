<!-- Solution to Question Nos 1 and 2 -->

Arithmetic Operators: + - * / % ** ++ --
<!-- Examples -->
let num1 = 2;
let num2 = 4;
console.log(num1 + num2);
console.log(num1 * num2);

Assignment Operators: = += -= *= /=
<!-- Examples -->
let num1 = 2;
let num1 += 20;

Comparison Operators: > < <= >= === == != !==
<!-- Examples -->
let num1 = 5;
let num2 = 2;
console.log(num1 == 5);
console.log(num2 === '2');

Logical Operators: && || !
<!-- Examples -->
const a = true; 
const b = false;
console.log(a && a); 
console.log(a || b);

Bitwise Operators: & | ^ ~ << >> >>>
<!-- Examples -->
let num1 = 12;
let num2 = 25;
console.log(num1 & num2);
console.log(num1 | num2);
